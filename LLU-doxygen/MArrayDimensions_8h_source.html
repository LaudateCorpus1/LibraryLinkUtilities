161  mint depth = checkContainerSize(std::distance(dimsBegin, dimsEnd));
162  auto dimsOk = std::all_of(dimsBegin, dimsEnd - 1, [](auto d) { return (d > 0) && (d <= (std::numeric_limits<mint>::max)()); }) &&
163  (dimsBegin[depth - 1] >= 0) && (dimsBegin[depth - 1] <= (std::numeric_limits<mint>::max)());
164  if (!dimsOk) {
165  ErrorManager::throwExceptionWithDebugInfo(ErrorName::DimensionsError, "Invalid input vector with array dimensions");
166  }
167  dims.reserve(depth);
